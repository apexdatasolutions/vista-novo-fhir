  class Condition extends Resource           -- Use to record detailed information about conditions, problems or diagnoses recognized by a clinician. There are many uses including: recording a Diagnosis during an Encounter; populating a problem List or a Summary Statement, such as a Discharge Summary.
  {
    attribute subject : Resource;            -- Subject of this condition.
    attribute encounter : Resource;          -- Encounter during which the condition was first asserted.
    attribute asserter : Resource;           -- Person who takes responsibility for asserting the existence of the condition as part of the electronic record.
    attribute dateAsserted : date;           -- Estimated or actual date the condition/problem/diagnosis was first detected/suspected.
    attribute code : CodeableConcept;        -- Identification of the condition, problem or diagnosis.
    attribute category : CodeableConcept;    -- A category assigned to the condition. E.g. finding | Condition | diagnosis | concern | condition.
    attribute status : code;                 -- The clinical status of the condition.
    attribute certainty : CodeableConcept;   -- The degree of confidence that this condition is correct.
    attribute severity : CodeableConcept;    -- A subjective assessment of the severity of the condition as evaluated by the clinician.
    attribute onset : ?;                     -- Estimated or actual date the condition began, in the opinion of the clinician.
    attribute abatement : ?;                 -- The date or estimated date that the condition resolved or went into remission. This is called "abatement" because of the many overloaded connotations associated with "remission" or "resolution" - Conditions are never really resolved, but they can abate.
    attribute stage : Condition_stage;       -- Clinical stage or grade of a condition. May include formal severity assessments.
    attribute evidence : Condition_evidence; -- Supporting Evidence / manifestations that are the basis on which this condition is suspected or confirmed.
    attribute location : Condition_location; -- The anatomical location where this condition manifests itself.
    attribute relatedItem : Condition_relatedItem; -- Further conditions, problems, diagnoses, procedures or events that are related in some way to this condition, or the substance that caused/triggered this Condition.
    attribute notes : String;                -- Additional information about the Condition. This is a general notes/comments entry  for description of the Condition, its diagnosis and prognosis.
  }

  class Condition_location                   -- The anatomical location where this condition manifests itself.
  {
    attribute code : CodeableConcept;        -- Code that identifies the structural location.
    attribute detail : String;               -- Detailed anatomical location information.
  }

  class Condition_evidence                   -- Supporting Evidence / manifestations that are the basis on which this condition is suspected or confirmed.
  {
    attribute code : CodeableConcept;        -- A manifestation or symptom that led to the recording of this condition.
    attribute detail : Resource;             -- Links to other relevant information, including pathology reports.
  }

  class Condition_relatedItem                -- Further conditions, problems, diagnoses, procedures or events that are related in some way to this condition, or the substance that caused/triggered this Condition.
  {
    attribute type : code;                   -- The type of relationship that this condition has to the related item.
    attribute code : CodeableConcept;        -- Code that identifies the target of this relationship. The code takes the place of a detailed instance target.
    attribute target : Resource;             -- Target of the relationship.
  }

  class Condition_stage                      -- Clinical stage or grade of a condition. May include formal severity assessments.
  {
    attribute summary : CodeableConcept;     -- A simple summary of the stage such as "Stage 3". The determination of the stage is disease-specific.
    attribute assessment : Resource;         -- Reference to a formal record of the evidence on which the staging assessment is based.
  }

  class ValueSet extends Resource            -- A value set specifies a set of codes drawn from one or more code systems.
  {
    attribute identifier : String;           -- The identifier that is used to identify this value set when it is referenced in a specification, model, design or an instance (should be globally unique OID, UUID, or URI).
    attribute version : String;              -- The identifier that is used to identify this version of the value set when it is referenced in a specification, model, design or instance. This is an arbitrary value managed by the profile author manually and the value should be a timestamp.
    attribute name : String;                 -- A free text natural language name describing the value set.
    attribute publisher : String;            -- The name of the individual or organization that published the value set.
    attribute telecom : Contact;             -- Contacts of the publisher to assist a user in finding and communicating with the publisher.
    attribute description : String;          -- A free text natural language description of the use of the value set - reason for definition, conditions of use, etc.
    attribute copyright : String;            -- A copyright statement relating to the value set and/or its contents.
    attribute status : code;                 -- The status of the value set.
    attribute experimental : boolean;        -- This valueset was authored for testing purposes (or education/evaluation/marketing), and is not intended to be used for genuine usage.
    attribute date : dateTime;               -- The date that the value set status was last changed.
    attribute define : ValueSet_define;      -- When value set defines its own codes.
    attribute compose : ValueSet_compose;    -- When value set includes codes from elsewhere.

    attribute code : code;                   -- Specifies a code or concept to be included or excluded.
    attribute filter : ValueSet_compose_include_filter; -- Select concepts by specify a matching criteria based on the properties (including relationships) defined by the system. If multiple filters are specified, they must all be true.
  }

    attribute concept : ValueSet_define_concept; -- Child Concepts (is-a / contains).

  class ValueSet_compose_include_filter      -- Select concepts by specify a matching criteria based on the properties (including relationships) defined by the system. If multiple filters are specified, they must all be true.

    attribute op : code;                     -- The kind of operation to perform as part of the filter criteria.
    attribute value : code;                  -- The match value may be either a code defined by the system, or a string value which is used a regex match on the literal string of the property value.
  }

  class ValueSet_expansion                   -- When value set is an expansion.

  class MedicationPrescription_dispense      -- Deals with details of the dispense part of the order.
  {
    attribute medication : Resource;         -- Identifies the medication that is to be dispensed.  This may be a more specifically defined than the medicationPrescription.medication . This is either a link to a resource representing the details of the medication or a simple attribute carrying a code that identifies the medication from a known list of medications.
    attribute validityPeriod : Period;       -- Design Comments: This indicates the validity period of a prescription (stale dating the Prescription) 
It reflects the prescriber perspective for the validity of the prescription. Dispenses must not be made against the prescription outside of this period. The lower-bound of the Dispensing Window signifies the earliest date that the prescription can be filled for the first time. If an upper-bound is not specified then the Prescription is open-ended or will default to a stale-date based on regulations. 
Rationale: Indicates when the Prescription becomes valid, and when it ceases to be a dispensable Prescription.
    attribute numberOfRepeatsAllowed : integer; -- An integer indicating the number of repeats of the Dispense. 
UsageNotes: For example, the number of times the prescribed quantity is to be supplied including the initial standard fill.
    attribute quantity : Quantity;           -- The amount that is to be dispensed.
    attribute expectedSupplyDuration : Duration; -- Identifies the period time over which the supplied product is expected to be used, or the length of time the dispense is expected to last. 
In some situations, this attribute may be used instead of quantity to identify the amount supplied by how long it is expected to last, rather than the physical quantity issued. E.g. 90 days supply of medication (based on an ordered dosage) When possible, it is always better to specify quantity, as this tends to be more precise. expectedSupplyDuration will always be an estimate that can be influenced by external factors.

  class Profile_structure_element            -- Captures constraints on each element within the resource.
  {
    attribute path : String;                 -- The path identifies the element and is expressed as a "."-separated list of ancestor elements, beginning with the name of the resource.
    attribute name : String;                 -- The name of this element definition (to refer to it from other element definitions using Profile.structure.element.definition.nameReference). This is a unique name referring to a specific set of constraints applied to this element. One use of this is to provide a name to different slices of the same element.
    attribute slicing : Profile_structure_element_slicing; -- Indicates that the element is sliced into a set of alternative definitions (there are multiple definitions on a single element in the base resource). The set of slices is any elements that come after this in the element sequence that have the same path, until a shorter path occurs (the shorter path terminates the set).
    attribute definition : Profile_structure_element_definition; -- Definition of the content of the element to provide a more specific definition than that contained for the element in the base resource.
  }

  class Profile_structure_element_slicing    -- Indicates that the element is sliced into a set of alternative definitions (there are multiple definitions on a single element in the base resource). The set of slices is any elements that come after this in the element sequence that have the same path, until a shorter path occurs (the shorter path terminates the set).
  {
    attribute discriminator : id;            -- Designates which child element is used to discriminate between the slices when processing an instance. The value of the child element in the instance must completely distinguish which slice the element in the resource matches based on the allowed values for that element in each of the slices.
    attribute ordered : boolean;             -- If the matching elements have to occur in the same order as defined in the profile.
    attribute rules : code;                  -- Whether additional slices are allowed or not. When the slices are ordered, profile authors can also say that additional slices are only allowed at the end.
  }

    attribute value : object;                -- Specifies a value that must hold for this element in the instance.
    attribute example : object;              -- An example value for this element.
    attribute maxLength : integer;           -- Indicates the shortest length that must be supported by conformant instances without truncation.
    attribute condition : id;                -- A reference to an invariant that may make additional statements about the cardinality in the instance.
    attribute constraint : Profile_structure_element_definition_constraint; -- Formal constraints such as co-occurrence and other constraints that can be computationally evaluated within the context of the instance.
    attribute mustSupport : boolean;         -- If true, conformant resource authors must be capable of providing a value for the element and resource consumers must be capable of extracting and doing something useful with the data element.  If false, the element may be ignored and not supported.

  class Patient_contact                      -- A contact party (e.g. guardian, partner, friend) for the patient.
  {
    attribute relationship : CodeableConcept; -- The nature of the relationship between the patient and the contact person.
    attribute name : HumanName;              -- A name associated with the person.
    attribute telecom : Contact;             -- A contact detail for the person, e.g. a telephone number or an email address.
    attribute address : Address;             -- Address for the contact person.
    attribute gender : CodeableConcept;      -- Administrative Gender - the gender that the person is considered to have for administration and record keeping purposes.
    attribute organization : Resource;       -- Organization on behalf of which the contact is acting or for which the contact is working.
  }

  class AdverseReaction extends Resource     -- Specific reactions to a substance.
  {
    attribute reactionDate : dateTime;       -- When the reaction occurred.
    attribute subject : Resource;            -- The subject of the adverse reaction.
    attribute didNotOccurFlag : boolean;     -- To say that a reaction to substance did not occur.
    attribute recorder : Resource;           -- Who recorded the reaction.
    attribute symptom : AdverseReaction_symptom; -- The signs and symptoms that were observed as part of the reaction.
    attribute exposure : AdverseReaction_exposure; -- An exposure to a substance that preceded a reaction occurrence.
  }

  class AdverseReaction_symptom              -- The signs and symptoms that were observed as part of the reaction.
  {
    attribute code : CodeableConcept;        -- Indicates the specific sign or symptom that was observed.
    attribute severity : code;               -- The severity of the sign or symptom.
  }

  class AdverseReaction_exposure             -- An exposure to a substance that preceded a reaction occurrence.
  {
    attribute exposureDate : dateTime;       -- When the exposure occurred.
    attribute exposureType : code;           -- Drug Administration, Immunization, Coincidental.
    attribute causalityExpectation : code;   -- A statement of how confident that the recorder was that this exposure caused the reaction.
    attribute substance : Resource;          -- Substance(s) that is presumed to have caused the adverse reaction.
  }

  class DiagnosticOrder extends Resource     -- A request for a diagnostic investigation service to be performed.
  {
    attribute subject : Resource;            -- Who or what the investigation is to be performed on. This is usually a human patient, but diagnostic tests can also be requested on animals, groups of humans or animals, devices such as dialysis machines, or even locations (typically for environmental scans).
    attribute orderer : Resource;            -- The practitioner that holds legal responsibility for ordering the investigation.
    attribute identifier : Identifier;       -- Identifiers assigned to this order by the order or by the receiver.
    attribute encounter : Resource;          -- An encounter that provides additional informaton about the healthcare context in which this request is made.
    attribute clinicalNotes : String;        -- An explanation or justification for why this diagnostic investigation is being requested.
    attribute specimen : Resource;           -- One or more specimens that the diagnostic investigation is about.
    attribute status : code;                 -- The status of the order.
    attribute priority : code;               -- The clinical priority associated with this order.
    attribute event : DiagnosticOrder_event; -- A summary of the events of interest that have occurred as the request is processed.
    attribute item : DiagnosticOrder_item;   -- The specific diagnostic investigations that are requested as part of this request. Sometimes, there can only be one item per request, but in most contexts, more than one investigation can be requested.
  }

  class DiagnosticOrder_event                -- A summary of the events of interest that have occurred as the request is processed.
  {
    attribute status : code;                 -- The status for the event.
    attribute date : dateTime;               -- The date/time at which the event occurred.
    attribute actor : Resource;              -- The person who was responsible for performing or recording the action.
  }

  class DiagnosticOrder_item                 -- The specific diagnostic investigations that are requested as part of this request. Sometimes, there can only be one item per request, but in most contexts, more than one investigation can be requested.
  {
    attribute code : CodeableConcept;        -- A code that identifies a particular diagnostic investigation that has been requested.
    attribute specimen : Resource;           -- If the item is related to a specific speciment.
    attribute bodySite : CodeableConcept;    -- Anatomical location where the request test should be performed.
    attribute status : code;                 -- The status of this individual item within the order.
    attribute event : DiagnosticOrder_event; -- A summary of the events of interest that have occurred as this item of the request is processed.
  }
